# Quick Commands Reference for Firebase Migration

## BACKEND COMMANDS

# Install dependencies
cd backend
npm install

# Start development server (with auto-reload)
npm run dev

# Start production server
npm start

# Create admin user
npm run bootstrap-admin

# Import authorized members from Excel
npm run import-members path/to/members.xlsx

# Import information data from Excel
npm run import-info path/to/information.xlsx

## FRONTEND COMMANDS

# Install dependencies
cd MyExpoApp
npm install

# Start Expo dev server
npm start

# Run on web
npm run web

# Run on iOS simulator
npm run ios

# Run on Android emulator
npm run android

## FIREBASE CLI COMMANDS

# Install Firebase CLI globally
npm install -g firebase-tools

# Login to Firebase
firebase login

# Initialize Firestore in project
firebase init firestore

# Deploy Firestore security rules
firebase deploy --only firestore:rules

# View Firestore logs
firebase firestore:logs

## CLEANUP MONGODB FILES

# Remove MongoDB configuration
Remove-Item -Path "backend\config\db.js" -Force

# Remove all Mongoose models
Remove-Item -Path "backend\models\*" -Force

# Uninstall Mongoose
cd backend
npm uninstall mongoose

## ENVIRONMENT SETUP

# Create .env file from template
cd backend
Copy-Item .env.example .env
# Then edit .env with your actual values

## TESTING

# Test backend health
curl http://localhost:3001/api/health

# Or in PowerShell
Invoke-WebRequest -Uri http://localhost:3001/api/health

## PRODUCTION DEPLOYMENT

# Build frontend for production
cd MyExpoApp
npm run build

# Deploy to Expo
npm run publish

## FIRESTORE OPERATIONS

# Backup Firestore data
firebase firestore:backup gs://your-bucket-name

# View Firestore usage
firebase firestore:stats

## TROUBLESHOOTING

# Clear npm cache
npm cache clean --force

# Reinstall all dependencies
cd backend
Remove-Item -Recurse -Force node_modules
Remove-Item -Force package-lock.json
npm install

# Check Firebase CLI version
firebase --version

# Check Node.js version
node --version

# Check npm version
npm --version

## REQUIRED FILES CHECKLIST

Backend:
- backend/serviceAccountKey.json (from Firebase Console)
- backend/.env (from .env.example)
- backend/firebase.js (already created)
- backend/config/firestore.js (already created)
- backend/firestore.rules (already created)

Frontend:
- MyExpoApp/firebaseConfig.ts (update with your config)
- MyExpoApp/config/api.ts (check BASE_URL)

## IMPORTANT NOTES

1. serviceAccountKey.json must be in backend/ folder
2. Never commit serviceAccountKey.json or .env to git
3. Update firebaseConfig.ts with your Firebase project details
4. Change admin password after first login
5. Deploy Firestore security rules before going to production
6. Test all features after migration
